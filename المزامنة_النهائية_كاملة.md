### 6. ✅ إصلاح منطق سلة التسوق
**في `js/products-new.js`:**

```javascript
window.handleAddToCart = function(productId) {
    const product = products.find(p => p.id === productId);
    
    // الحصول على السلة من localStorage
    let cart = JSON.parse(localStorage.getItem('media_cart')) || [];
    
    // التحقق من وجود المنتج في السلة
    const existingItem = cart.find(item => item.id === productId);
    
    if (existingItem) {
        existingItem.quantity = (existingItem.quantity || 1) + 1;
    } else {
        cart.push({
            id: product.id,
            name: product.name,
            price: product.price,
            image: product.image || getProductImage(product.category, 0),
            category: product.category,
            quantity: 1
        });
    }
    
    // الحفظ في localStorage
    localStorage.setItem('media_cart', JSON.stringify(cart));
    
    // تحديث العداد
    updateCartCounter();
    
    // إظهار Toast
    showToast(`Added ${product.name} to cart!`);
};
```

**بنية كائن السلة:**

```javascript
    category: "laptop",
    quantity: 2
}
```

### 7. ✅ البحث العام يعمل
**في `js/main.js`:**

```javascript
function setupGlobalSearch() {
    const globalSearch = document.getElementById('globalSearch');
    
    function handleSearch(searchTerm) {
        if (searchTerm.trim()) {
            window.location.href = `products.html?search=${encodeURIComponent(searchTerm)}`;
        }
    }
    
    if (globalSearch) {
        globalSearch.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') {
                handleSearch(e.target.value);
            }
        });
    }
}
```

**في `js/products-new.js`:**

```javascript
// التحقق من مصطلح البحث من URL
const urlSearch = getUrlParameter('search');
if (urlSearch) {
    searchTerm = urlSearch.toLowerCase();
    const searchInput = document.getElementById('searchInput');
    if (searchInput) {
        searchInput.value = urlSearch;
    }
}

// فلترة المنتجات
if (searchTerm) {
    filtered = filtered.filter(p => 
        p.name.toLowerCase().includes(searchTerm) ||
        p.category.toLowerCase().includes(searchTerm) ||
        (p.description && p.description.toLowerCase().includes(searchTerm))
    );
}
```
